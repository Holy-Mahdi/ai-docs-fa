گرادیان کاهشی (Gradient Descent)

۱. ایده کلی

گرادیان کاهشی یک روش تکراری برای پیدا کردن مینیمم یک تابع هزینه است.

۲. فرمول

θ ← θ - α ∇θ J(θ)

۳. پیاده‌سازی ساده (Python)

import numpy as np

def gradient_descent(X, y, theta, lr=0.01, epochs=1000):
    m = len(y)
    for i in range(epochs):
        preds = X.dot(theta)
        error = preds - y
        grad = (1/m) * X.T.dot(error)
        theta = theta - lr * grad
    return theta
